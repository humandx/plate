### Installation

```bash npm2yarn
npm install @humandx/plate
# or
npm install @humandx/plate-find-replace
npm install @humandx/plate-ui-find-replace
```

### Usage

import { FindReplaceSandpack } from './FindReplaceSandpack';

<FindReplaceSandpack />

### Options

```tsx
export interface FindReplacePlugin {
  /**
   * Searching text to highlight
   */
  search?: string;
}
```
